CREATE USER : 
		GRANT ALL PRIVILAGES ON <database_name>.* TO '<username>'@'localhost' IDENTIFIED BY '<password'>;
		Imp: <database_name>.* ===> Only given db and all its object
		     <database_name>.object ===> Only given db and given object
